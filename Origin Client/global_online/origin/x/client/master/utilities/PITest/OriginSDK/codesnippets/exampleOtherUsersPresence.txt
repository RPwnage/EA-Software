void exampleOtherUsersPresence()
{
   const int NUM_USERS = 5;
   OriginUserT users[NUM_USERS];

   users[0] = 342342390;
   users[1] = 48209384342;
   users[2] = 4829384932940;
   users[3] = 724837490;
   users[4] = 47289374893;

   // Similarly this game may want to know what other users are doing. This is accomplished using two techniques:
   // This queries a user's presence status directly.
   OriginHandleT handle;
   OriginErrorT err = OriginQueryPresence(OriginGetDefaultUser(), users, NUM_USERS, PresenceResultCallback, NULL, 10000, &handle);

   // This subscribes to changes in a user's presence status.
   err = OriginSubscribePresence(OriginGetDefaultUser(), users, NUM_USERS);
}
