void exampleSynchronousCall()
{
   OriginProfileT profile;
   OriginHandleT handle;
   OriginErrorT err;

   // Call a synchronous function. The application receives the data as soon as the function completes.
   if(( err = OriginGetProfileSync(0, 10000, &profile, &handle)) == ORIGIN_SUCCESS)
   {
       printf("I am %s, my User Id is %I64d\n", profile.PersonaName, profile.UserId);
       std::cout << "I am " << profile.PersonaName << ", my User Id is " << profile.UserId << std::endl;
   }
   // By destroying the handle, the data associated with it becomes invalid.
   err = OriginDestroyHandle(handle);
}
