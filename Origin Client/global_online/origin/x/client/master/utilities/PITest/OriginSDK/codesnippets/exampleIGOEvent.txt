void exampleIGOEvent()
{
   // Register for the IGO event. This will normally form part of the initial set up for the SDK.
   OriginRegisterEventCallback(ORIGIN_EVENT_IGO, IgoEventCallback, NULL);

   // When we display the In-Game Overlay, this will generate an event that will result in the
   // IgoEventCallback being called.
   OriginErrorT err = OriginShowIGO(true);
}
