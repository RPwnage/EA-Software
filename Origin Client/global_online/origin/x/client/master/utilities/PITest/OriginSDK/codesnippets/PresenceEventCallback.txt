// This is the callback for SDK Presence Event handling.
OriginErrorT PresenceEventCallback (int32_t eventId, void* pContext, void* eventData, uint32_t count)
{
   OriginPresenceT *event = (OriginPresenceT *)eventData;
   if (event)
   {
       printf("Presence EventId: %d, User: %I64u, PresenceState: %d, Presence: %S, GamePresence: %S", 
           eventId, 
           event->UserId, 
           event->PresenceState, 
           event->Presence, 
           event->GamePresence);
   }
   return ORIGIN_SUCCESS;
}
