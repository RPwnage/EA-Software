void exampleFriendsIGO()
{
   // Display the IGO Show Friends dialog. This is the IGO equivalent to the programatic OriginQueryFriendsSync call.
   OriginErrorT err = OriginShowFriendsUI(OriginGetDefaultUser());

   // A user to which to send a friend invite.
   const OriginUserT USER = 454551247836;

   // Display the Show Friends IGO dialog.
   err = OriginRequestFriendUI(OriginGetDefaultUser(), USER);

   // Show the profile of a particular friend.
   err = OriginShowFriendsProfileUI(OriginGetDefaultUser(), USER);
}
