void exampleSetGetPresence()
{
   enumPresence ePresence = ORIGIN_PRESENCE_OFFLINE;
   // This sets the default user's presence information.
   OriginErrorT err = OriginSetPresence(OriginGetDefaultUser(), ePresence, "Battlefield 3", "example game presence", "4283409235004523");

   enumPresence presence;
   char RichPresence[255];
   char GamePresence[255];
   char Session[128];

   // The game can check this user's presence as follows:
   err = OriginGetPresence(OriginGetDefaultUser(), &presence, RichPresence, 255, GamePresence, 255, Session, 128);
   if(err == ORIGIN_SUCCESS)
   {
       printf("GetPresence: %d - %S - %S - %S", presence, RichPresence, GamePresence, Session);
   }
}
