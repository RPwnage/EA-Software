void exampleProfile()
{
   OriginProfileT profile;
   OriginHandleT handle;

   // The user is us, so this has to be zero.
   const uint32_t thisUser = 0;
   OriginErrorT err = OriginGetProfileSync(thisUser, 10000, &profile, &handle);
   printf("User: %I64u  AvatarId:%s", profile.UserId, profile.AvatarId);
   // It is important to destroy the handle to release the resource.
   err = OriginDestroyHandle(handle);

   // To check if our user is online use the OriginCheckOnline function:
   int8_t online;
   if((err = OriginCheckOnline(&online)) == ORIGIN_SUCCESS)
       if (online) printf("You are online");
}
