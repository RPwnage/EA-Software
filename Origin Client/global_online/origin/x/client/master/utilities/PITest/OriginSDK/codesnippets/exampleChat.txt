void exampleChat()
{
   // The user that the chat message will be sent to.
   const OriginUserT USER = 454551247836;

   // Register for a callback to receive chat events from the other user.
   OriginRegisterEventCallback(ORIGIN_EVENT_CHAT, ChatEventCallback, NULL);

   // Initiate the chat session from within the game.
   char Thread[20];
   char Body[255];
   sprintf_s(Thread, 20, "%S", "ChitChat Thread");
   sprintf_s(Body, 255, "%S", "Bla Bla Bla");
   OriginErrorT err = OriginSendChat(OriginGetDefaultUser(), USER, Thread, Body);
}
