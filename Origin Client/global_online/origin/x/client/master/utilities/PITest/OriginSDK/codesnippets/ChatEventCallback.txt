// Callback for SDK Chat Event handling
OriginErrorT ChatEventCallback (int32_t eventId, void* pContext, void* eventData, uint32_t count)
{
   OriginChatT *event = (OriginChatT *)eventData;
   if (event)
       printf("Chat EventId: %d, From: %I64d, Thread: %S, Message: %S", eventId, event->From, event->Thread, event->Message);

   return ORIGIN_SUCCESS;
}
