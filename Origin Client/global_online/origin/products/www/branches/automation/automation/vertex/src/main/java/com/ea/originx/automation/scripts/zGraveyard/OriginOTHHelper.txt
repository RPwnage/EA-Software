package com.ea.originx.automation.lib.helpers.csv;

/**
 * Represents each row in oth_info.csv file.
 *
 * @author nvarthakavi
 */
public class OriginOTHHelper {

    protected String entitlementName;
    protected String entitlementOfferId;
    protected String countryCode;
    protected String languageCode;
    protected String entitlementPartialPDPUrl;
    protected String startTime;
    protected String endTime;

    /**
     * Constructor for OriginOTHHelper
     *
     * @param entitlementOfferId       OfferID of the entitlement
     * @param entitlementName          Name of the entitlement
     * @param countryCode              Name of the country (country code - 'abc')
     * @param languageCode             Language used by the country ( language code - 'ab-cd')
     * @param entitlementPartialPDPUrl Partial PDP URL of the entitlement
     * @param startTime                Valid start time for the entitlement in OTH (MM/DD/YYY HH:MM AM PDT)
     * @param endTime                  Valid end time for the entitlement in OTH (MM/DD/YYY HH:MM AM PDT)
     */
    public OriginOTHHelper(String entitlementName, String entitlementOfferId, String countryCode, String languageCode, String entitlementPartialPDPUrl, String startTime, String endTime) {
        this.entitlementName = entitlementName;
        this.entitlementOfferId = entitlementOfferId;
        this.countryCode = countryCode;
        this.languageCode = languageCode;
        this.entitlementPartialPDPUrl = entitlementPartialPDPUrl;
        this.startTime = startTime;
        this.endTime = endTime;
    }

    /**
     * Gets the 'On the House' entitlement name.
     *
     * @return Name of the entitlement
     */
    public String getEntitlementName() {
        return entitlementName;
    }

    /**
     * Gets the 'On the House' entitlement offerID.
     *
     * @return OfferId of the entitlement
     */
    public String getEntitlementOfferId() {
        return entitlementOfferId;
    }

    /**
     * Gets the 'On the House' entitlement's partial PDP URL.
     *
     * @return Partial PDP URL of the entitlement
     */
    public String getEntitlementPartialPDPUrl() {
        return entitlementPartialPDPUrl;
    }

    /**
     * Gets the country code.
     *
     * @return the country code.
     */
    public String getCountryCode() {
        return countryCode;
    }

    /**
     * Gets the language code.
     *
     * @return the language code.
     */
    public String getLanguageCode() {
        return languageCode;
    }

    /**
     * Gets valid start time of the 'On the House' entitlement.
     *
     * @return start time of the 'On the House' entitlement.
     */
    public String getEntitlementStartTime() {
        return startTime;
    }

    /**
     * Gets valid end time of the 'On the House' entitlement.
     *
     * @return end time of the 'On the House' entitlement.
     */
    public String getEntitlementEndTime() {
        return endTime;
    }
}
