import org.testng.ITestContext;

/**
 * Tests the 'On the House' Entitlements in Country 'can' (Canada) and Language 'fr-ca' (French)
 *
 * @author nvarthakavi
 */
public class OAOnTheHouseCANfr_ca extends OAOnTheHouse {

    @Test(groups = {"rtb", "rtb_oth", "client_only"})
    public void testOnTheHouseCANfr_ca(ITestContext context) throws Exception {
        String testName = MethodHandles.lookup().lookupClass().toString().split(" ")[1];
        testOnTheHouse(context, CountryInfo.CountryEnum.CANADA, LanguageInfo.LanguageEnum.FRENCH_CANADA, testName);
    }
}
