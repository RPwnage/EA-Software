	OriginIGORenderer *m_igoRenderer = new OriginIGORenderer();

	OriginIGORenderingContext_CALLBACK_T igoContext = { 0 };
	igoContext.base.size = sizeof(OriginIGORenderingContext_CALLBACK_T);
	igoContext.base.rendererType = ORIGIN_IGO_RENDERER_CALLBACK;
	igoContext.base.executionType = static_cast<enumIGOExecutionType>(ORIGIN_IGO_CAPTURE_VIDEO | ORIGIN_IGO_RENDER_IGO);

	}
