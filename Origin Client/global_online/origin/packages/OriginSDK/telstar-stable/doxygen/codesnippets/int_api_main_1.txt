// ////////////////////////////////////////////////////////////////////

// //////////////////////////////////////////////////////////////////// 
void OriginSDKErrorCallback(void * pcontext, OriginErrorT error, const char* message_text, const char* file, int line)
{
    char msgString[4096] = { 0 };
    sprintf_s(msgString, _countof(msgString), "%s(%i): %s\n", file, line, message_text);
    OutputDebugStringA(msgString);
}


// ////////////////////////////////////////////////////////////////////

// current IGO API callback
bool gIsIGOVisible = false;

// OriginIGOVisibilityCallbackFunc
void OriginIGOVisibilityCallback(bool *visible)
{
    if (!visible)
        return;

    gIsIGOVisible = *visible;
}
