	OriginErrorT OriginIGORenderer::streamingSetup(const OriginIGOStreamingInfo_T* input)
	{
		// Setup streaming properties
		
		if (input != nullptr)
		{
			m_streamingProperties.height = input->height;
			m_streamingProperties.width = input->width;
			m_streamingProperties.format = input->format;
			m_streamingProperties.fps = input->fps;
			m_streamingProperties.streaming = input->streaming;
		}

		return ORIGIN_SUCCESS_IGOAPI;
	}
