	OriginErrorT OriginIGORenderer::deleteTexture2D(int32_t inputTextureId)
	{
		if (inputTextureId != enumOriginIGOTextureID::ORIGINIGOTEXTUREID_INVALID)
		{
			if (inputTextureId == m_streamData.texID)
				m_streamData.texID = enumOriginIGOTextureID::ORIGINIGOTEXTUREID_INVALID;

			TextureMap::iterator it = m_textures.find(inputTextureId);
			if (it != m_textures.end())
			{
				m_textures.erase(it);
				return ORIGIN_SUCCESS_IGOAPI;
			}
		}
		return ORIGIN_SUCCESS_IGOAPI;
	}
