void exampleProfileIGO()
{  
   // A user about which to display their profile.
   const OriginUserT USER = 454551247836;

   // This is how to display the profile of the user of this PC.
   OriginErrorT err = OriginShowProfileUI(OriginGetDefaultUser());

   // This is how to display the profile of a friend whose user id we know.
   err = OriginShowFriendsProfileUI(OriginGetDefaultUser(), USER);
}
