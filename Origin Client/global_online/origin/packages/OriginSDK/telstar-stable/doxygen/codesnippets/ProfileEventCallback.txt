// Callback for SDK Profile Event handling
OriginErrorT ProfileEventCallback (int32_t eventId, void* pContext, void* eventData, uint32_t count)
{
   uint32_t *event = (uint32_t *)eventData;
   if (event)
   {
       printf("Profile EventId: %d, EventData: %d", eventId, *event);
   }
   return ORIGIN_SUCCESS;
}
