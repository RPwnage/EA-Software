// This callback is invoked when we are invited by a friend to join their game.
OriginErrorT PendingInviteEventCallback (int32_t eventId, void* pContext, void* eventData, uint32_t count)
{
   OriginPendingInviteT *invite = (OriginPendingInviteT *)eventData;
   if (invite)
   {
       printf("Invite EventId: %d, FromId: %I64d, MultiplayerId: %S" ,
           eventId, //Should be ORIGIN_EVENT_INVITE_PENDING in this case
           invite->FromId, //Nucleus Id of user who sent the invite
           invite->MultiplayerId); //Game identifier
   }
   return ORIGIN_SUCCESS;
}

void examplePendingInviteEvent()
{
   // Invites from our friends can be received by registering for the pending invite event.
   // This would normally be performed at start up, but is shown here for clarity.
   OriginRegisterEventCallback(ORIGIN_EVENT_INVITE_PENDING, PendingInviteEventCallback, NULL);
}
