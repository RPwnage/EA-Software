// Callback for SDK Friends Event handling
OriginErrorT FriendsEventCallback (int32_t eventId, void* pContext, void* eventData, uint32_t count)
{
   OriginFriendT *friends = (OriginFriendT *)eventData;
   if (friends)
       printf("Friends EventId: %d, Number of Friends: %d", eventId, count);
   return ORIGIN_SUCCESS;
}
