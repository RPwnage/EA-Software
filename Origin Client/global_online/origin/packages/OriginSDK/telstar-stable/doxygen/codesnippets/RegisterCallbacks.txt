OriginErrorT RegisterCallbacks(void * pContext)
{
        // Register for SDK Events
    OriginRegisterEventCallback(~0x0, EventCallback, pContext);
    OriginRegisterEventCallback(ORIGIN_EVENT_IGO, IgoEventCallback, pContext);
    OriginRegisterEventCallback(ORIGIN_EVENT_INVITE, InviteEventCallback, pContext);
    OriginRegisterEventCallback(ORIGIN_EVENT_LOGIN, LoginEventCallback, pContext);
    OriginRegisterEventCallback(ORIGIN_EVENT_PROFILE, ProfileEventCallback, pContext);
    OriginRegisterEventCallback(ORIGIN_EVENT_PRESENCE, PresenceEventCallback, pContenxt);
    OriginRegisterEventCallback(ORIGIN_EVENT_FRIENDS, FriendsEventCallback, pContext);
    OriginRegisterEventCallback(ORIGIN_EVENT_PURCHASE, PurchaseEventCallback, pContext);
    OriginRegisterEventCallback(ORIGIN_EVENT_CONTENT, ContentEventCallback, pContext);
    OriginRegisterEventCallback(ORIGIN_EVENT_BLOCKED,  BlockListEventCallback, pContext);
    OriginRegisterEventCallback(ORIGIN_EVENT_CHAT_MESSAGE,  ChatEventCallback, pContext);
    OriginRegisterEventCallback(ORIGIN_EVENT_ONLINE,  OnlineStatusEventCallback, pContext);
    OriginRegisterEventCallback(ORIGIN_EVENT_ACHIEVEMENT,  AchievementEventCallback, pContext);
    OriginRegisterEventCallback(ORIGIN_EVENT_INVITE_PENDING, InvitePendingEventCallback, pContext);
    return ORIGIN_SUCCESS;
}
