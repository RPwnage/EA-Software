44 // xenon friend id (xuid) is a 64-bit value which turns out to be a 20-char base-10 numeral string
45 // ps3 friend id is always set to "unavailable"
46 #define FRIENDAPI_FRIENDID_MAXLEN (20)
47 
48 // xenon friend name is filled with the gamertag which is limited to 15 chars (XUSER_MAX_NAME_LENGTH) in the XONLINE_FRIEND struct
49 // ps3 friend name is filled with the PSN online id, which is a 16-char string (see SceNpId struct)
50 #define FRIENDAPI_FRIENDNAME_MAXLEN (16) // max string length for friend name
51 
52 // xenon: currently not used
53 // ps3: matches value of SCE_NP_BASIC_MAX_MESSAGE_SIZE (maximum message data size)
54 #define FRIENDAPI_EVENTMSG_MAXSIZE (512)
55 
56 // xenon: MS Game Developer support confirmed that XBOX360 currently does not support more than 100 friends per gamer
57 // ps3: PS3 doc mentions that the number of friends per gamer is limited to 100 (See NP_Basic-Reference_e.pdf)
58 #define FRIENDAPI_MAX_FRIENDS (100)
59 
60 // ps3 blocked user id is filled with the PSN online id, which is a 16-char string (see SceNpId struct)
61 #define FRIENDAPI_BLOCKEDUSERID_MAXLEN (16) // max string length for blocked user id
62 
63 // ps3: PS3 doc mentions that the maximum entry count of the block list is 50 (See NP_Common-Reference_e.pdf)
